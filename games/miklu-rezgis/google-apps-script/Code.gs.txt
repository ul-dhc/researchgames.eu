const SESSIONS_SHEET = 'sessions';
const EVALUATIONS_SHEET = 'evaluations';
const SPREADSHEET_ID = '1P4vS-YtPka5V3l4W8Jlro5jFiF-Y6wjmzsLJYqb9Qpk';
const SPREADSHEET_TIME_ZONE = 'Europe/Riga';

const SESSION_HEADERS = [
  'session_id', 'session_started_at', 'last_updated_at', 'session_finished_at',
  'session_status', 'game_version', 'participation_type', 'research_consent',
  'planned_riddle_count', 'completed_riddle_count', 'last_completed_riddle',
  'total_time_ms', 'total_points', 'time_bonus_points', 'solved_riddles',
  'given_up_riddles', 'hints_used', 'wrong_attempts', 'shuffle_count',
  'age_group', 'first_time_player', 'game_rating', 'player_comment',
  'device_type', 'screen_size_group', 'theme', 'font_size', 'daily_replay',
  'feedback_submitted', 'app_version', 'data_schema_version'
];

const EVALUATION_HEADERS = [
  'response_id', 'session_id', 'played_at', 'riddle_order', 'riddle_id',
  'riddle_level', 'game_version', 'participation_type', 'party_player_slot',
  'riddle_text', 'answer', 'riddle_location', 'riddle_collection', 'solved',
  'gave_up', 'time_ms', 'points', 'time_bonus_points', 'hints_used',
  'wrong_attempts', 'shuffle_count', 'heard_before', 'knew_answer_before_search',
  'solution_method', 'difficulty_rating',
  'knows_similar_variant', 'similar_variant_text', 'variant_source',
  'research_questions_skipped', 'app_version', 'data_schema_version',
  'unit_number'
];

function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu('Mīklu režģis')
    .addItem('Initialize analytics sheets', 'initializeSheets')
    .addToUi();
}

function initializeSheets() {
  const spreadsheet = getSpreadsheet_();
  spreadsheet.setSpreadsheetTimeZone(SPREADSHEET_TIME_ZONE);
  ensureSheet_(spreadsheet, SESSIONS_SHEET, SESSION_HEADERS);
  ensureSheet_(spreadsheet, EVALUATIONS_SHEET, EVALUATION_HEADERS);
  formatTimestampColumns_(spreadsheet.getSheetByName(SESSIONS_SHEET), SESSION_HEADERS);
  formatTimestampColumns_(spreadsheet.getSheetByName(EVALUATIONS_SHEET), EVALUATION_HEADERS);
}

function getSpreadsheet_() {
  return SpreadsheetApp.openById(SPREADSHEET_ID);
}

function doGet() {
  initializeSheets();
  const spreadsheet = getSpreadsheet_();
  return jsonResponse_({
    ok: true,
    service: 'Mīklu režģis analytics',
    spreadsheet_id: spreadsheet.getId(),
    spreadsheet_name: spreadsheet.getName(),
    schema_version: '1',
    timestamp: new Date().toISOString()
  });
}

function doPost(event) {
  const lock = LockService.getScriptLock();

  try {
    lock.waitLock(20000);
    initializeSheets();

    const body = parseRequest_(event);
    const records = Array.isArray(body.records) ? body.records : [body];

    if (!records.length) {
      throw new Error('No records supplied.');
    }

    const results = records.map(processRecord_);
    SpreadsheetApp.flush();

    return jsonResponse_({
      ok: true,
      processed: results.length,
      results: results
    });
  } catch (error) {
    console.error(error);
    return jsonResponse_({ ok: false, error: String(error.message || error) });
  } finally {
    try {
      lock.releaseLock();
    } catch (error) {
      // The lock may not have been acquired if parsing failed immediately.
    }
  }
}

function processRecord_(rawRecord) {
  const record = rawRecord && rawRecord.data ? rawRecord.data : rawRecord;
  const recordType = String(
    rawRecord.record_type || rawRecord.type || record.record_type || record.type || ''
  ).toLowerCase();

  if (recordType === 'evaluation' || recordType === 'riddle_evaluation') {
    return upsertEvaluation_(record);
  }

  if (
    recordType === 'session' ||
    recordType === 'session_update' ||
    recordType === 'session_feedback'
  ) {
    return upsertSession_(record);
  }

  throw new Error('Unknown record_type: ' + recordType);
}

function upsertSession_(record) {
  if (!record.session_id) {
    throw new Error('session_id is required for session records.');
  }

  const sheet = getSpreadsheet_().getSheetByName(SESSIONS_SHEET);
  const cleanRecord = normalizeRecord_(record, SESSION_HEADERS);
  cleanRecord.last_updated_at = cleanRecord.last_updated_at || new Date().toISOString();

  const rowNumber = findRowById_(sheet, 1, cleanRecord.session_id);

  if (rowNumber) {
    mergeRow_(sheet, rowNumber, SESSION_HEADERS, cleanRecord);
    return { record_type: 'session', id: cleanRecord.session_id, action: 'updated' };
  }

  appendRecord_(sheet, SESSION_HEADERS, cleanRecord);
  return { record_type: 'session', id: cleanRecord.session_id, action: 'created' };
}

function upsertEvaluation_(record) {
  if (!record.response_id) {
    throw new Error('response_id is required for evaluation records.');
  }
  if (!record.session_id) {
    throw new Error('session_id is required for evaluation records.');
  }

  const sheet = getSpreadsheet_().getSheetByName(EVALUATIONS_SHEET);
  const cleanRecord = normalizeRecord_(record, EVALUATION_HEADERS);
  cleanRecord.played_at = cleanRecord.played_at || new Date().toISOString();

  const rowNumber = findRowById_(sheet, 1, cleanRecord.response_id);

  if (rowNumber) {
    mergeRow_(sheet, rowNumber, EVALUATION_HEADERS, cleanRecord);
    return { record_type: 'evaluation', id: cleanRecord.response_id, action: 'updated' };
  }

  appendRecord_(sheet, EVALUATION_HEADERS, cleanRecord);
  return { record_type: 'evaluation', id: cleanRecord.response_id, action: 'created' };
}

function parseRequest_(event) {
  if (!event || !event.postData || !event.postData.contents) {
    throw new Error('Empty POST body.');
  }

  try {
    return JSON.parse(event.postData.contents);
  } catch (error) {
    throw new Error('POST body must be valid JSON.');
  }
}

function ensureSheet_(spreadsheet, name, headers) {
  let sheet = spreadsheet.getSheetByName(name);
  if (!sheet) {
    sheet = spreadsheet.insertSheet(name);
  }

  const existingHeaders = sheet.getLastColumn() > 0
    ? sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]
    : [];

  const headersMatch = headers.every((header, index) => existingHeaders[index] === header);
  if (!headersMatch) {
    sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
  }

  sheet.setFrozenRows(1);
  sheet.getRange(1, 1, 1, headers.length)
    .setFontWeight('bold')
    .setBackground('#0f766e')
    .setFontColor('#ffffff');

  return sheet;
}

function normalizeRecord_(record, headers) {
  const normalized = {};

  headers.forEach(function(header) {
    if (Object.prototype.hasOwnProperty.call(record, header)) {
      const value = record[header];
      if (isTimestampHeader_(header) && value) {
        const parsedDate = new Date(value);
        normalized[header] = isNaN(parsedDate.getTime()) ? safeCellValue_(value) : parsedDate;
      } else {
        normalized[header] = safeCellValue_(value);
      }
    }
  });

  return normalized;
}

function isTimestampHeader_(header) {
  return header === 'played_at' || /_at$/.test(header);
}

function formatTimestampColumns_(sheet, headers) {
  headers.forEach(function(header, index) {
    if (isTimestampHeader_(header)) {
      sheet.getRange(2, index + 1, Math.max(sheet.getMaxRows() - 1, 1), 1)
        .setNumberFormat('dd.MM.yyyy HH:mm:ss');
    }
  });
}

// Run once from the Apps Script editor if older timestamp cells were stored as ISO text.
function convertExistingTimestamps() {
  const spreadsheet = getSpreadsheet_();
  spreadsheet.setSpreadsheetTimeZone(SPREADSHEET_TIME_ZONE);
  [
    { sheet: spreadsheet.getSheetByName(SESSIONS_SHEET), headers: SESSION_HEADERS },
    { sheet: spreadsheet.getSheetByName(EVALUATIONS_SHEET), headers: EVALUATION_HEADERS }
  ].forEach(function(config) {
    const sheet = config.sheet;
    if (!sheet || sheet.getLastRow() < 2) return;
    config.headers.forEach(function(header, index) {
      if (!isTimestampHeader_(header)) return;
      const range = sheet.getRange(2, index + 1, sheet.getLastRow() - 1, 1);
      const converted = range.getValues().map(function(row) {
        const value = row[0];
        if (!value || value instanceof Date) return [value];
        const parsedDate = new Date(value);
        return [isNaN(parsedDate.getTime()) ? value : parsedDate];
      });
      range.setValues(converted).setNumberFormat('dd.MM.yyyy HH:mm:ss');
    });
  });
}

function safeCellValue_(value) {
  if (value === null || typeof value === 'undefined') return '';
  if (typeof value === 'boolean' || typeof value === 'number') return value;
  if (typeof value === 'object') value = JSON.stringify(value);

  const text = String(value);
  return /^[=+\-@]/.test(text) ? "'" + text : text;
}

function findRowById_(sheet, idColumn, id) {
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return 0;

  const match = sheet
    .getRange(2, idColumn, lastRow - 1, 1)
    .createTextFinder(String(id))
    .matchEntireCell(true)
    .findNext();

  return match ? match.getRow() : 0;
}

function appendRecord_(sheet, headers, record) {
  const row = headers.map(function(header) {
    return Object.prototype.hasOwnProperty.call(record, header) ? record[header] : '';
  });
  sheet.appendRow(row);
}

function mergeRow_(sheet, rowNumber, headers, record) {
  const range = sheet.getRange(rowNumber, 1, 1, headers.length);
  const current = range.getValues()[0];

  const merged = headers.map(function(header, index) {
    return Object.prototype.hasOwnProperty.call(record, header)
      ? record[header]
      : current[index];
  });

  range.setValues([merged]);
}

function jsonResponse_(payload) {
  return ContentService
    .createTextOutput(JSON.stringify(payload))
    .setMimeType(ContentService.MimeType.JSON);
}
